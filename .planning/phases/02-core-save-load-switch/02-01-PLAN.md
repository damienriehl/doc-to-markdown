    **Exported helper (for testing):**

    ```javascript
    export function buildSnapshot(book, chapters) {
      return JSON.stringify({
        book,
        chapters: chapters.map(({ file, _dragging, ...rest }) => rest),
      });

    ```

    **Hook shape:**
    ```javascript
    export function useProjectStore() {
      const [book, setBook] = useState({ title: "", author: "" });
      const [chapters, setChapters] = useState([]);
      const [activeProjectId, setActiveProjectId] = useState(null);
      const [activeProjectName, setActiveProjectName] = useState("");
      const [projectList, setProjectList] = useState([]);
      const [bootStatus, setBootStatus] = useState("idle");   // "idle" | "loading" | "ready"
      const [saveStatus, setSaveStatus] = useState("saved");   // "saved" | "unsaved" | "saving"
      const savedSnapshotRef = useRef(null);
      const loadTokenRef = useRef(0);
      const pendingSwitchRef = useRef(null);
    ```

    **Boot hydration effect (runs once on mount):**
    ```javascript
    useEffect(() => {
      async function boot() {
        setBootStatus("loading");
        try {
          const list = await listProjects();
          setProjectList(list);
          const lastId = getLastProjectId();
          if (lastId) {
            const record = await getProject(lastId);
            if (record) {
              const blobMap = await getFiles(lastId);
              const { book: b, chapters: c } = deserializeProject(record, blobMap);
              setBook(b);
              setChapters(c);
              setActiveProjectId(lastId);
              setActiveProjectName(record.name);
              savedSnapshotRef.current = buildSnapshot(b, c);
            }
          }
        } finally {
          setBootStatus("ready");
        }
      }
      boot();
    }, []);
    ```

    **isDirty derived value:**
    ```javascript
    const isDirty = useMemo(() => {
      if (!savedSnapshotRef.current) {
        return chapters.length > 0 || book.title !== "";
      }
      return buildSnapshot(book, chapters) !== savedSnapshotRef.current;
    }, [book, chapters]);
    ```

    **saveStatus sync effect:**
    ```javascript
    useEffect(() => {
      setSaveStatus(prev => {
        if (prev === "saving") return prev; // don't interrupt in-flight save
        return isDirty ? "unsaved" : "saved";
      });
    }, [isDirty]);
    ```

    **beforeunload guard:**
    ```javascript
    useEffect(() => {
      if (!isDirty) return;
      const handler = (e) => { e.preventDefault(); e.returnValue = ""; };
      window.addEventListener("beforeunload", handler);
      return () => window.removeEventListener("beforeunload", handler);
    }, [isDirty]);
    ```

    **save function:**
    ```javascript
    const save = useCallback(async (name) => {
      const id = activeProjectId || crypto.randomUUID();
      const projectName = name || activeProjectName || "Untitled";
      setSaveStatus("saving");
      try {
        const { projectRecord, blobs } = serializeProject({
          id, name: projectName, book, chapters, uiState: {},
          createdAt: activeProjectId ? undefined : undefined, // let serializer generate
        });
        await putProject(projectRecord);
        if (blobs.length > 0) await putFiles(id, blobs);
        saveLastProjectId(id);
        await requestPersistentStorage();
        setActiveProjectId(id);
        setActiveProjectName(projectName);
        savedSnapshotRef.current = buildSnapshot(book, chapters);
        setSaveStatus("saved");
        // Refresh project list
        const list = await listProjects();
        setProjectList(list);
        return id;
      } catch (err) {
        setSaveStatus("unsaved");
        throw err;
      }
    }, [book, chapters, activeProjectId, activeProjectName]);
    ```

    **load function (with load-sequence token):**
    ```javascript
    const load = useCallback(async (id) => {
      const token = ++loadTokenRef.current;
      const record = await getProject(id);
      if (!record) return null;
      const blobMap = await getFiles(id);
      if (token !== loadTokenRef.current) return null; // stale
      const { book: b, chapters: c } = deserializeProject(record, blobMap);
      setBook(b);
      setChapters(c);
      setActiveProjectId(id);
      setActiveProjectName(record.name);
      saveLastProjectId(id);
      savedSnapshotRef.current = buildSnapshot(b, c);
      setSaveStatus("saved");
      return { book: b, chapters: c };
    }, []);
    ```

    **switchProject (isDirty guard):**
    ```javascript
    const switchProject = useCallback(async (id) => {
      if (isDirty) {
        pendingSwitchRef.current = id;
        return { blocked: true, pendingId: id };
      }
      return load(id);
    }, [isDirty, load]);

    const confirmSwitch = useCallback(async () => {
      const id = pendingSwitchRef.current;
      if (!id) return null;
      pendingSwitchRef.current = null;
      return load(id);
    }, [load]);

    const cancelSwitch = useCallback(() => {
      pendingSwitchRef.current = null;
    }, []);
    ```

    **newProject (reset to blank state):**
    ```javascript
    const newProject = useCallback(() => {
      setBook({ title: "", author: "" });
      setChapters([]);
      setActiveProjectId(null);
      setActiveProjectName("");
      savedSnapshotRef.current = null;
      setSaveStatus("saved");
    }, []);
    ```

    **Return value:**
    ```javascript
    return {
      book, setBook, chapters, setChapters,
      activeProjectId, activeProjectName, projectList,
      isDirty, saveStatus, bootStatus,
      save, load, switchProject, confirmSwitch, cancelSwitch, newProject,
    };
    ```

    Key implementation notes:
    - Set `savedSnapshotRef.current` using values from `deserializeProject` return (NOT from React state, which hasn't updated yet) — prevents isDirty false-positive flicker after load

    - Boot effect must always reach `setBootStatus("ready")` via try/finally — even on first run with no projects
    - `loadTokenRef` prevents stale async results from overwriting a freshly switched project
