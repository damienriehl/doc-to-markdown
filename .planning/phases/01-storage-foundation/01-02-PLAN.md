    **Imports:**

    **Constants:**
    ```javascript
    export const DB_VERSION = 1;
    const DB_NAME = "doc-to-markdown";
    ```

    **Singleton pattern:**
    ```javascript
    let _db = null;

    function getDb() {
      if (_db) return _db;

      const db = new Dexie(DB_NAME);

      db.version(DB_VERSION).stores({
        projects: "id, name, updatedAt",
        files: "id, projectId",
      });

      db.on("versionchange", () => {
        db.close();
      });

      db.on("blocked", () => {
        console.warn("[projectDb] IndexedDB upgrade blocked -- close other tabs to proceed.");
      });

      _db = db;
      return db;
    }
    ```

    **Test helper (exported for test cleanup):**
    ```javascript
    export async function _resetDbForTest() {
      if (_db) {
        _db.close();
        await _db.delete();
        _db = null;
      }
    }
    ```

    **Project CRUD section** (with `// --- Project CRUD ---` separator):

    `putProject(projectRecord)`:
    - Try `await getDb().projects.put(projectRecord)`
    - Catch: if `err.name === "QuotaExceededError"`, throw `new Error("Storage full. Free up browser storage and try again.")`
    - Otherwise rethrow

    `getProject(id)`:
    - Return `getDb().projects.get(id)`

    `listProjects()`:
    - Return `getDb().projects.orderBy("updatedAt").reverse().toArray()`

    `deleteProject(id)`:
    - Use `getDb().transaction("rw", [getDb().projects, getDb().files], async () => { ... })`
    - Inside: `await getDb().projects.delete(id)` and `await getDb().files.where("projectId").equals(id).delete()`

    **File Blob Storage section** (with `// --- File Blob Storage ---` separator):

    `putFiles(projectId, blobs)`:
    - Map blobs to records: `{ id, projectId, blob: file, name, size: file.size, type: file.type, lastModified: file.lastModified }`
    - Try `await getDb().files.bulkPut(records)`
    - Catch QuotaExceededError same as putProject

    `getFiles(projectId)`:
    - Query `getDb().files.where("projectId").equals(projectId).toArray()`
    - Build `new Map()`, for each record: `map.set(rec.id, new File([rec.blob], rec.name, { type: rec.type, lastModified: rec.lastModified }))`
    - Return the Map

    **Storage Persistence section** (with `// --- Storage Persistence ---` separator):

    `requestPersistentStorage()`:
    - Try `navigator.storage?.persist?.()` with await
    - Return the boolean result, or `false` if unavailable
    - Never throw
